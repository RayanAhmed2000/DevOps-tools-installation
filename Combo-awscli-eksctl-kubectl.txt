#!/bin/bash

# Exit immediately if a command exits with a non-zero status
set -e

echo "---------------------------------------"
echo "Starting Installation of AWS Tools"
echo "---------------------------------------"

# 1. AWS CLI
echo "Installing AWS CLI now..."
sudo apt update && sudo apt install -y unzip curl
curl "https://awscli.amazonaws.com/awscli-exe-linux-x86_64.zip" -o "awscliv2.zip"
unzip -q awscliv2.zip
sudo ./aws/install --update
rm -rf awscliv2.zip ./aws
aws --version

# 2. Kubectl
echo "Installing kubectl now..."
K8S_VERSION=$(curl -L -s https://dl.k8s.io/release/stable.txt)
curl -LO "https://dl.k8s.io/release/${K8S_VERSION}/bin/linux/amd64/kubectl"
chmod +x kubectl
sudo mv kubectl /usr/local/bin/
kubectl version --client

# 3. EKSctl
echo "Installing EKSctl now..."
# Downloads the latest amd64 release for Linux
curl -sL "https://github.com/eksctl-io/eksctl/releases/latest/download/eksctl_$(uname -s)_amd64.tar.gz" | tar xz -C /tmp
sudo mv /tmp/eksctl /usr/local/bin
eksctl version

echo "---------------------------------------"
echo "All tools installed in /usr/local/bin"
echo "---------------------------------------"
